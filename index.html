CREATE TABLE AI_Healthcare (
    id INTEGER PRIMARY KEY,
    application_name VARCHAR(50),
    description TEXT,
    category VARCHAR(30),
    accuracy FLOAT,
    data_source VARCHAR(50),
    year_of_introduction INTEGER,
    usage_count INTEGER,
    user_rating FLOAT,
    last_updated DATE
);

# Bar Chart of Applications by Category
import matplotlib.pyplot as plt

# Bar Chart
plt.figure(figsize=(10, 6))
category_data = %sql SELECT category, COUNT(*) as application_count FROM AI_Healthcare GROUP BY category;
categories = [row[0] for row in category_data]
counts = [row[1] for row in category_data]

plt.bar(categories, counts, color='#156082')
plt.xlabel('Category')
plt.ylabel('Applications')
plt.xticks(rotation=45)
plt.show()

# Line Chart
plt.figure(figsize=(10, 6))
ratings_data = %sql SELECT ROUND(user_rating, 1) as rating, COUNT(*) as count FROM AI_Healthcare GROUP BY rating ORDER BY rating;
ratings = [row[0] for row in ratings_data]
counts = [row[1] for row in ratings_data]

plt.plot(ratings, counts, marker='o', linestyle='-', color='#156082', linewidth=2, markersize=8)
plt.title('Ratings for AI Healthcare Applications')
plt.xlabel('Rating')
plt.ylabel('Applications')
plt.xticks(ratings)
plt.grid(True)
plt.show()

# Line Chart for Year of Introduction Trends
plt.figure(figsize=(10, 6))
trends_data = %sql SELECT year_of_introduction, COUNT(*) as application_count FROM AI_Healthcare GROUP BY year_of_introduction ORDER BY year_of_introduction;
years = [row[0] for row in trends_data]
counts = [row[1] for row in trends_data]

plt.plot(years, counts, marker='o', linestyle='-', color='#156082', linewidth=2, markersize=8)
plt.title('Trends in Introduction of AI Healthcare Applications (2015-2023)', fontsize=16, fontweight='bold')
plt.xticks(years, fontsize=12)
plt.grid(True, linestyle='--', alpha=0.7)
plt.tight_layout() 
plt.show()
